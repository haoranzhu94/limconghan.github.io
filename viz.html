def extract_mirp_features(mirp):
    rates = [p_data.rate for p_data in mirp.ports.values()]
    feature_vec = rates
    return feature_vec

def preprocess(n, data):
    matrix = np.zeros((n,n))
    minus_me = np.zeros((n,n))
    for i in range(n):
        for j in range(n):
            matrix[i,j] = data['baseline'][i] - list(data['seeded'][i][j])[0]
    return matrix
